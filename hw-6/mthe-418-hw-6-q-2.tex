    \begin{solution}
      \begin{proof}
        Let \(\bm{v} \in L\). By the property of \(L\), \(\exists \epsilon > 0\) such that
        \begin{equation*}
          L \cap \{\bm{w} \in \R^{m} : \norm{\bm{w}} < \epsilon\} = \{\bm{0}\}.
        \end{equation*}
        Then for \(\bm{v} = \bm{0} \in L\), it follows that
        \begin{equation*}
          L \cap \{\bm{w} \in \R^{m} : \norm{\bm{v - w}} < \epsilon\} = \{\bm{v}\}.
        \end{equation*}
      \end{proof}
    \end{solution}
